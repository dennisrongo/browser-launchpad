Extensibility Demonstration: Adding a New Theme

Example code showing how to add a "Midnight Blue" theme:


.midnight-blue {
  --color-primary: #1E40AF;
  --color-background: #0F172A;
  --color-surface: #1E293B;
  --color-text: #F1F5F9;
  --color-text-secondary: #94A3B8;
  --color-border: #334155;
}


export interface Settings {
  id: string
  theme: 'modern-light' | 'dark-elegance' | 'midnight-blue'  // Add new option
  grid_columns: number
  grid_gap: number
  created_at: string
  updated_at: string
}





if (newSettings.theme === 'dark-elegance' || newSettings.theme === 'midnight-blue') {
  document.documentElement.classList.add('dark')
} else {
  document.documentElement.classList.remove('dark')
}


const themeClassMap: Record = {
  'modern-light': '',
  'dark-elegance': 'dark',
  'midnight-blue': 'midnight-blue'
}
document.documentElement.className = themeClassMap[newSettings.theme] || ''
