<project_specification>
  <project_name>Browser Launchpad</project_name>

  <overview>
    A modern Chrome extension that replaces the new tab page with a customizable, widget-based dashboard.
    Users can create multiple pages, add various widgets (bookmarks, weather, AI chat, clock), customize the layout,
    choose themes, and export/import their data. The extension features a clean, modern interface and supports
    multiple AI providers for chat functionality.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React + TypeScript</framework>
      <styling>Tailwind CSS</styling>
      <build_tool>Vite</build_tool>
      <architecture>Widget-based component architecture for extensibility</architecture>
    </frontend>
    <backend>
      <runtime>Chrome Extension Manifest v3</runtime>
      <database>Chrome Storage API (local storage)</database>
      <export_format>JSON</export_format>
    </backend>
    <communication>
      <api>OpenAI API for AI chat</api>
      <api>Straico API for AI chat with dynamic model fetching</api>
      <weather_api>OpenWeatherMap API (or similar)</weather_api>
    </communication>
  </technology_stack>

  <prerequisites>
    <environment_setup>
      - Node.js and npm/pnpm installed
      - Chrome browser for development and testing
      - Chrome Extension Developer Mode enabled
      - API keys for OpenAI and/or Straico (user-provided)
      - Weather API key (user-provided for weather widget)
    </environment_setup>
  </prerequisites>

  <database_schema>
    <tables>
      <pages>
        - id: string (unique identifier)
        - name: string (page name)
        - order: number (for page reordering)
        - widgets: array of widget objects
        - created_at: timestamp
        - updated_at: timestamp
      </pages>
      <widgets>
        - id: string (unique identifier)
        - type: string ('bookmark', 'weather', 'ai-chat', 'clock')
        - page_id: string (reference to parent page)
        - order: number (for widget reordering)
        - title: string (widget display title)
        - config: object (widget-specific configuration)
        - created_at: timestamp
      </widgets>
      <settings>
        - id: string ('global-settings')
        - theme: string (selected theme name)
        - grid_columns: number (1-6)
        - grid_gap: number (spacing in pixels)
        - ai_providers: array of provider configurations
        - created_at: timestamp
        - updated_at: timestamp
      </settings>
      <ai_provider_config>
        - provider: string ('openai', 'straico')
        - api_key: string (encrypted)
        - selected_model: string
        - enabled: boolean
      </ai_provider_config>
      <chat_history>
        - id: string (unique identifier)
        - widget_id: string (reference to AI chat widget)
        - messages: array of message objects
        - model: string (model used)
        - provider: string (provider used)
        - created_at: timestamp
      </chat_history>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <Chrome_Storage_API>
      - chrome.storage.local.get()

      - chrome.storage.local.remove()
      - chrome.storage.local.clear()
      - chrome.storage.onChanged listener
    </Chrome_Storage_API>
    <AI_Providers>
      - POST https://api.openai.com/v1/chat/completions
      - GET https://api.straico.com/v2/models
      - POST https://api.straico.com/v0/chat/completions
    </AI_Providers>
    <Weather_API>
      - GET https://api.openweathermap.org/data/2.5/weather
      - Query parameters: q={city}, appid={api_key}
    </Weather_API>
    <Bookmark_Metadata>
      - GET {url} (fetch page title for bookmarks)
    </Bookmark_Metadata>
  </api_endpoints_summary>
